  <script>
    const grid = document.getElementById('grid');
    const finalizar = document.getElementById('finalizar');
    const totalNumeros = 200;
    const numeroWhatsApp = '5521964508216';
    const mensagemInicio = 'Olá! Quero confirmar minha participação na Rifa Solidária Redmi 11 PRO 5G. Meus números escolhidos são:';
    const vendidos = []; // adicione aqui os números já vendidos, ex: [5,23,40]

    const selecionados = new Set();

    for(let i=1;i<=totalNumeros;i++){
      const btn=document.createElement('div');
      btn.textContent=i;
      btn.classList.add('num');
      if(vendidos.includes(i)) btn.classList.add('sold');
      btn.addEventListener('click',()=>{
        if(btn.classList.contains('sold')) return;
        if(selecionados.has(i)) selecionados.delete(i);
        else selecionados.add(i);
        btn.classList.toggle('selected');
      });
      grid.appendChild(btn);
    }

    finalizar.addEventListener('click',()=>{
      if(selecionados.size===0){alert('Selecione ao menos um número.');return;}
      const nums=[...selecionados].sort((a,b)=>a-b).join(', ');
      const msg=`${mensagemInicio} ${nums}.`;
      const link=`https://wa.me/${numeroWhatsApp}?text=${encodeURIComponent(msg)}`;
      window.open(link,'_blank');
    });
  </script>
